<script>
   const vm = Vue.createApp({
     data() {
		return {
			quantity : 100,
			price :100
		}
	  },
	  computed : {
		subtotalComputed : function () {
			console.log('computed');
			return this.quantity * this.price;
		}
	  },
	  methods : {
		subtotalMethods : function () {
			console.log('methods');
			return this.quantity * this.price;
		}
	  }
	}) .mount('#app');
  </script>
 </body>    	
</html>
